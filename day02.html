函数的三种定义方式

FUNCTIONS

函数是 JavaScript 中可重复使用的代码块。JS 有三种主要定义方式，各有优劣：

方式 语法示例 提升 (Hoisting) this 绑定 建议
函数声明 function foo() {} ✅ 整体提升 动态（调用时决定） ✅ 通用首选
函数表达式 const foo = function() {} ❌ 只提升变量 动态（调用时决定） 需赋给变量时
箭头函数 const foo = () => {} ❌ 不提升 静态（词法作用域） ✅ 回调/简短函数

📝 方式一：函数声明

最经典的写法，会被提升 (Hoisting)——可以在定义之前调用。

// ✅ 可以在声明前调用（提升）
greet("张三");

function greet(name) {
  console.log(`你好，${name}！`);
}

greet("李四");  // 同样可以调用

📦 方式二：函数表达式

把函数赋值给变量。不会整体提升——只有变量声明被提升。

// ❌ 不能在赋值前调用
// sayHi("张三");  // ReferenceError: Cannot access 'sayHi' before initialization

const sayHi = function(name) {
  console.log(`Hi, ${name}!`);
};

sayHi("张三");  // ✅ 定义后调用没问题

⚡ 方式三：箭头函数 (ES6+)

现代 JS 中最常用的写法。简洁、不绑定自己的 this，非常适合回调。

// 完整写法
const add = (a, b) => {
  return a + b;
};

// 简写：只有一个表达式时，可省略 {} 和 return
const add2 = (a, b) => a + b;

// 只有一个参数时，可省略 ()
const square = x => x * x;

// 无参数时，必须写 ()
const sayHello = () => console.log("Hello!");

// 返回对象字面量：需要用 () 包裹
const makePoint = (x, y) => ({ x, y });

console.log(add(3, 5));       // 8
console.log(square(4));        // 16
console.log(makePoint(10, 20)); // {x: 10, y: 20}
💡
选择建议：现代代码中，箭头函数是最流行的默认选择。但当你需要 this 绑定（如对象方法）或用 arguments 对象时，仍需要用普通函数。今天先记住它们的区别，后面会深入。

参数：默认值、剩余参数、解构

PARAMETERS

🎯 默认参数值 (ES6+)

调用时如果没传某个参数，自动使用默认值。

function createMarker(name, lng = 116.4, lat = 39.9, height = 0) {
  console.log(`📍 ${name}：经度 ${lng}，纬度 ${lat}，高度 ${height}m`);
}

createMarker("天安门");                       // 后三个参数用默认值
createMarker("外滩", 121.49, 31.23);         // 覆盖前两个默认值
createMarker("珠峰", 86.92, 27.98, 8848); // 全部覆盖

📦 剩余参数 (Rest Parameters, ES6+)

用 ...args 捕获不定数量的参数为一个真数组，替代旧的 arguments 对象。

function sum(...numbers) {
  let total = 0;
  for (const n of numbers) {
    total += n;
  }
  return total;
}

console.log(sum(1, 2, 3));       // 6
console.log(sum(10, 20));          // 30
console.log(sum());                 // 0（空数组）

// 前面可以有固定参数，...rest 必须是最后一个
function log(prefix, ...messages) {
  console.log(`[${prefix}]`, ...messages);
}

log("Cesium", "加载地形数据...", "完成", 200);
// [Cesium] 加载地形数据... 完成 200

🔓 参数解构 (ES6+)

直接从参数位置解构对象或数组，尤其适合配置对象。

// 对象解构参数（CesiumJS 中最常见的模式）
function initViewer({ container, zoom = 12, center = [0, 0], enableLighting = true }) {
  console.log(`容器：${container}，缩放：${zoom}，中心点：[${center}]`);
}

// 只传必要参数
initViewer({ container: "cesiumContainer" });
// 容器：cesiumContainer，缩放：12，中心点：[0,0]

// 完整配置
initViewer({
  container: "map",
  zoom: 18,
  center: [116.4, 39.9],
  enableLighting: false
});
ℹ️
关于 arguments：普通函数内部有一个类数组对象 arguments（箭头函数没有）。但现代代码中，...rest 是更好的选择，因为它返回的是真数组，可以直接用 forEach、map 等方法。

返回值与提前返回

RETURN

↩️ return 语句

函数通过 return 把结果"交还"给调用者。一旦执行 return，函数立即停止。

function multiply(a, b) {
  return a * b;
  // 这行永远不会执行 ↓
  console.log("不会打印");
}

const result = multiply(6, 7);
console.log(result);  // 42

🚫 无 return = undefined

如果函数没有 return 语句，则自动返回 undefined。

function doSomething() {
  console.log("执行了");
  // 没有 return
}

const val = doSomething();
console.log(val);  // undefined

🛡️ 提前返回 (Early Return)

提前 return 可以让代码更扁平，避免深层嵌套——这是推荐的编码风格。

// ❌ 深层嵌套（不推荐）
function getZoomLevel(type) {
  if (type) {
    if (type === "city") {
      return 14;
    } else if (type === "building") {
      return 18;
    } else {
      return 10;
    }
  }
  return 5; // 默认值
}

// ✅ 提前返回 + 扁平化（推荐）
function getZoomLevel2(type) {
  if (!type) return 5;           // 提前返回默认值
  if (type === "city") return 14;
  if (type === "building") return 18;
  return 10;
}

作用域：全局、函数与块级

SCOPE

作用域决定了在哪里可以访问变量。JavaScript 有三层作用域：

🌐 全局作用域

在所有函数和块之外声明的变量，整个程序任何位置都能访问。

const APP_NAME = "CesiumEarth";  // 全局变量

function showApp() {
  console.log(APP_NAME);  // ✅ 可以访问
}

showApp();
console.log(APP_NAME);     // ✅ 也能访问
⚠️
注意：全局变量过多会造成"命名污染"，应尽量减少使用。

🏠 函数作用域

在函数内部声明的变量，只能在函数内部访问。

function initMap() {
  const apiKey = "abc123";  // 函数作用域内的变量
  console.log(apiKey);      // ✅ 可以访问
}

initMap();
// console.log(apiKey);  ❌ ReferenceError: apiKey is not defined

📐 块级作用域 (ES6+)

用 let/const 声明的变量，只在最近的 {} 块内有效。这是 var 不具备的特性。

// let/const：块级作用域
{
  const x = 10;
  let y = 20;
  console.log(x, y);  // ✅ 10 20
}
// console.log(x, y);  ❌ ReferenceError

// var：没有块级作用域！
{
  var z = 30;
}
console.log(z);  // ✅ 30（var 泄漏到块外！）

// for 循环也是块级作用域
for (let i = 0; i < 3; i++) {
  // i 只在这个块内有效
}
// console.log(i);  ❌ ReferenceError
💡
这就是为什么不用 var！ var 无视 {} 边界，极易导致意外的变量覆盖和泄漏。坚持 const/let，让作用域规则简单明了。

作用域链与闭包入门

CLOSURE

🔗 作用域链

当函数访问一个变量时，JavaScript 从当前作用域开始查找，逐级向上，直到全局作用域。

const globalVar = "我在全局";

function outer() {
  const outerVar = "我在外层";

  function inner() {
    const innerVar = "我在内层";

    console.log(innerVar);  // ✅ 找到自己
    console.log(outerVar);  // ✅ 向上找到外层
    console.log(globalVar); // ✅ 继续向上找到全局
  }

  inner();
  // console.log(innerVar);  ❌ 外层无法访问内层变量
}

outer();

🔐 闭包 (Closure) - 入门

闭包 = 函数 + 它能访问的外部变量。当内部函数被返回并在外部调用时，它依然能记住它诞生时的环境。

// 经典例子：创建"计数器"函数
function createCounter() {
  let count = 0;  // 这个变量会被闭包"记住"

  return function() {
    count++;
    return count;
  };
}

const counter1 = createCounter();
console.log(counter1());  // 1
console.log(counter1());  // 2
console.log(counter1());  // 3

// 每次调用 createCounter() 创建独立的闭包
const counter2 = createCounter();
console.log(counter2());  // 1（独立计数）
ℹ️
闭包在 CesiumJS 中无处不在：事件回调、材质函数、瓦片加载器都依赖闭包来保持状态。理解闭包是理解和调试大型 JS 应用的基础。

立即执行函数 (IIFE)

IIFE

📦 什么是 IIFE？

IIFE = Immediately Invoked Function Expression。定义后立即执行一次，且不会污染全局作用域。

// 经典 IIFE 语法
(function() {
  const privateVar = "外部无法访问";
  console.log("IIFE 执行了！");
  console.log(privateVar);
})();

// 带参数的 IIFE
(function(name) {
  console.log(`Hello, ${name}!`);
})("CesiumJS");

// 箭头函数版 IIFE（ES6+）
(() => {
  console.log("箭头函数的 IIFE");
})();

// console.log(privateVar);  ❌ ReferenceError — 隔离成功！

🎯 现代替代：块级作用域

在 ES6+ 中，很多 IIFE 的用途可以被 块级作用域替代：

// 旧式：用 IIFE 隔离变量
(function() {
  const temp = "我不污染外部";
  // 做一些事...
})();

// 现代：直接用块级作用域 {} + let/const
{
  const temp = "我也不污染外部";
  // 做一些事...
}
// console.log(temp); ❌ ReferenceError — 同样隔离！

不过 IIFE 仍有价值：当你需要异步初始化（如 async IIFE）或用 var 的旧代码中隔离作用域时。

🌍 CesiumJS 关联场景

CESIUM

函数与作用域在 CesiumJS 开发中无处不在。以下是你将来会遇到的真实场景：

🗺️ 场景 1：定义工具函数

CesiumJS 大量使用角度和弧度，你经常会写这样的转换函数：

// 度数 ↔ 弧度（CesiumJS 内部使用弧度）
const toRadians  = (deg) => deg * Math.PI / 180;
const toDegrees  = (rad) => rad * 180 / Math.PI;

// 飞往指定位置（参数解构）
function flyTo({ lng, lat, alt = 5000, duration = 2 }) {
  // viewer.camera.flyTo({ ... })  将来会调用 Cesium API
  console.log(`✈️ 飞往 [${lng},${lat}]，高度 ${alt}m，耗时 ${duration}s`);
}

flyTo({ lng: 116.4, lat: 39.9 });          // alt 和 duration 用默认值
flyTo({ lng: 121.5, lat: 31.2, alt: 10000 }); // 覆盖高度

📡 场景 2：事件回调 / 闭包捕获

CesiumJS 中大量使用回调函数处理用户交互和瓦片加载：

// 点击地球表面添加标记（闭包捕获 viewer 引用）
function setupClickHandler(viewer) {
  // 这个回调函数是闭包：它"记住"了 viewer
  const handler = (position) => {
    // viewer.entities.add({ position, ... })
    console.log(`点击位置：${JSON.stringify(position)}`);
  };

  // viewer.screenSpaceEventHandler.setInputAction(handler, ...)
  console.log("✅ 点击处理器已设置（闭包捕获了 viewer）");
  return handler;
}

setupClickHandler({ name: "myViewer" });

🎨 场景 3：材质与着色器函数

自定义 CesiumJS 材质时，你需要编写加工函数：

// 创建一个带透明度的颜色函数（闭包保持透明度值）
function createAlphaColor(alpha = 0.5) {
  // 返回的函数"记住"了 alpha 的值
  return (r, g, b) => {
    return `rgba(${r}, ${g}, ${b}, ${alpha})`;
  };
}

const semiTransparent = createAlphaColor(0.3);
console.log(semiTransparent(100, 200, 50));  // "rgba(100, 200, 50, 0.3)"

📋 今日知识点总结

  • ✅ JavaScript 有三种函数定义方式：函数声明（可提升）、函数表达式、箭头函数（最简洁）
  • ✅ 箭头函数不绑定自己的 this，不能用作构造函数，非常适合回调
  • ✅ 默认参数用 = 设置，剩余参数用 ...args 捕获
  • ✅ 参数解构让配置对象传参变得优雅，是 CesiumJS 中最常见的模式
  • ✅ 作用域有三层：全局、函数、块级（let/const）
  • ✅ var 没有块级作用域，会导致泄漏——坚持用 const/let
  • ✅ 闭包 = 函数 + 它能访问的外部变量，内部函数"记住"诞生时的环境
  • ✅ IIFE 定义后立即执行，不污染全局；现代代码中块级作用域常可替代它
  • ✅ 提前返回（Early Return）让代码更扁平、更易读
